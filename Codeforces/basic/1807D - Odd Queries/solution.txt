t = int(input())
for _ in range(t):
    n, q = map(int, input().split())
    a = list(map(int, input().split()))
    total = sum(a)
    prefix = [0] * (n + 1)
    for i in range(n):
        prefix[i + 1] = prefix[i] + a[i]
    for i in range(q):
        l ,r,k = map(int, input().split())
        current_sum = total - (prefix[r] - prefix[l - 1]) + (r - l + 1) * k
        if current_sum % 2 != 0:
            print("YES")
        else:
            print("NO")
