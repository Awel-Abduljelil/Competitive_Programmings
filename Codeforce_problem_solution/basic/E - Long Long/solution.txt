t = int(input())

for _ in range(t):
    n = int(input())
    arr = list(map(int, input().split()))

    total = sum(abs(x) for x in arr)

    count = 0
    neg = False

    for x in arr:
        if x < 0:
            if not neg:
                count += 1
            neg = True
        elif x > 0:
            neg = False

    print(total, count)
