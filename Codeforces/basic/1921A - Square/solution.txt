#include <bits/stdc++.h>
using namespace std;

int main() {
    int rawf;
    cin >> rawf;
    
    while (rawf--) {
        vector<pair<int, int>> pts(4);
        for (int i = 0; i < 4; i++) {
            cin >> pts[i].first >> pts[i].second;
        }
        
        int side = 0;
        for (int i = 1; i < 4; i++) {
            if (pts[i].first != pts[0].first) {
                side = abs(pts[i].first - pts[0].first);
                break;
            }
        }
        
        cout << side * side << "\n";
    }
    
    return 0;
}
